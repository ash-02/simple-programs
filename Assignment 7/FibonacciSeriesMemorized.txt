def FibonacciSeriesMemorized_Aux(n, r):
        if r[n] > -1:
                return r[n]
        if n == 0:
                q = 0
        else if n == 1:
                q = 1
        else:

                                + FibonacciSeriesMemorized_Aux(n - 2, r)
