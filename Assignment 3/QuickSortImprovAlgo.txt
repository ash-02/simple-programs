PARTITION`(A, p, r)
    x = A[r]
    i = p - 1
    t = p - 1

    for j = p to r - 1
        if A[j] < x:
            i = i + 1
            t = t + 1
            exchange A[i] with A[j]
        else if A[j] == x:
            t = t + 1
            exchange A[t] with A[j]

    exchange A[i + 1] with A[r]
    return i + 1, t
